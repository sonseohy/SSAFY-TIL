### 2차원 배열

```python
# 1차원 배열 생성
arr1 = [0] * 3

# 2차원 배열 생성
arr2 = [[0] * 3 for _ in range(2)]

# 1차원 배열 출력
print(arr1)
print(*arr1)

# 2차원 배열 출력 1
for i in range(2):
    print(arr2[i])

# 2차원 배열 출력 2
for i in range(2):
    for j in range(3):
        print(arr2[i][j], end=' ')
    print()
```

```python
# 주의
arr = [[0] * 3] * 2
print(arr)

arr[0][0] = 1
print(arr) # arr[1][0]도 같이 바뀜
```
- 이중 for문에서 i는 행을 나타낼 때 주로 사용, j는 열을 나타낼 때 주로 사용

```python
# 열 우선 순회
for j in range(m):
```

#### 연습문제 1-2
```python
N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]

di = [0, 1, 0, -1]
dj = [1, 0, -1, 0]

total = 0

for i in range(N):
    for j in range(N):  # NxN 배열의 모든 원소에 대해
        s = 0   #문제에서 원소와 주변 인접 원소의 차의 절대값의 합
        # i, j 원소의 4방향 원소에 대해
        for k in range(4):
            ni = i + di[k]
            nj = j + dj[k]
            if 0 <= ni < N and 0 <= nj < N:
                s += abs(arr[i][j] - arr[ni][nj])    # 실존하는 인접 원소 ni, nj
        total += s

print(total)
```

```python
# 방법 2
for row in range(N):

        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            newR = row + dr
            newC = col + dc
        if 0 <= newR < N and 0 <= newC < N:
            sum_v += abs(arr[newR][newC] - arr[row][col])
```

#### 각 행의 합 중 최댓값과 인덱스 출력
```python
'''
3 5
45 15 10 56 23
96 98 99 40 69
96 84 49 46 34
'''

N, M = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(N)]

max_r = 0
for row in range(N):
    sum_v = 0
    for col in range(M):
        sum_v += arr[row][col]
        if max_r < sum_v:
            max_r = sum_v
            max_idx = row
    print(sum_v)
print(max_r, max_idx)
```

#### 바깥 모서리에 있는 값의 합 구하기
```python
'''
3 5
45 15 10 56 23
96 98 99 40 69
96 84 49 46 34
'''

N, M = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(N)]

sum_v = 0
row = 0
for col in range(M):
    sum_v += arr[row][col]

row = N - 1
for col in range(M):
    sum_v += arr[row][col]

col = 0
for row in range(N):
    sum_v += arr[row][col]

sum_v = sum_v - arr[0][0] - arr[0][M-1] - arr[N-1][0] - arr[N-1][M-1]

print(sum_v)
```

#### 연습문제2 : 부분집합 합 문제
```python
arr = [-7, -5, 2, 3, 8, -2, 4, 6, 9]
n = len(arr)

for i in range(1<<n):
    sum_v = 0
    for j in range(n):
        if i & (1<<j):
            sum_v += arr[j]
print(sum_v)
```
